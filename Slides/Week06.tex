\frame{
	\frametitle{Survey Mode}
	\begin{itemize}\itemsep2em
		\item Definition?
		\item<2-> Manner in which survey responses are collected
		\item<3-> Not necessarily related to sampling frame
		\item<4-> A survey can use multiple modes
	\end{itemize}
}

\frame{
	\frametitle{Modes}
	\begin{itemize}\itemsep1em
		\item Personal interviewing
		\begin{itemize}
            \item Traditional personal interviewing
    		\item Computer-assisted personal interviewing (CAPI)
		\end{itemize}
		\item Self-administered
    	\begin{itemize}
            \item Paper-and-pencil
    		\item Computer-assisted self-interview (CASI)
    		\item Interactive Voice Response (IVR)
    		\item Telephone-Audio Computer Assisted Self-Interviewing (T-ACASI)
    		\item Web-based
        \end{itemize}
        \item New modes
        \begin{itemize}
            \item Web-based IVR or avatars
            \item Mobile
            \item SMS
        \end{itemize}
	\end{itemize}
}

        		\item<2-> Journals, logs, biometric data
    		\end{itemize}
		\item<3-> Why would we mix multiple modes?
	\end{itemize}
}

\section{Costs and Budgeting}
\frame{\tableofcontents[currentsection]}

\frame{
	\frametitle{Three Types of Costs}
	\begin{itemize}\itemsep2em
    	\item Fixed costs
    	\item Variable costs
    	\item Incidental costs (contingencies)
	\end{itemize}
}

\frame{
	\frametitle{Fixed Costs}
	\begin{itemize}\itemsep2em
    	\item Regardless of sample size
    	\item Examples
        	\begin{itemize}
            	\item Cost of creating/finding/building frame
            	\item Sampling plan development
            	\item Staff salaries
            	\item Infrastructure
            	\item Questionnaire development
            	\item Interviewer training
            	\item Data cleaning and analysis
            	\item Lottery-style respondent incentives
        	\end{itemize}
	\end{itemize}
}

\frame{
	\frametitle{Variable Costs}
	\begin{itemize}\itemsep2em
    	\item Depends on sample size and response rate
        	\begin{enumerate}
            	\item Per-attempt
            	\item Per-interview
        	\end{enumerate}
    	\item Examples
        	\begin{itemize}
            	\item Interviewer wage
            	\item Postage
            	\item Materials (e.g., questionnaires)
            	\item Travel time
            	\item Per-response respondent incentives
        	\end{itemize}
	\end{itemize}
}

\frame{
	\frametitle{Incidental Costs}
	\begin{itemize}\itemsep2em
    	\item Contingency planning (may not spend)
    	\item Examples
        	\begin{itemize}
            	\item Weather
            	\item Lost or stolen data or materials
            	\item Technology failures
        	\end{itemize}
	\end{itemize}
}


\frame{\frametitle{Questions about survey budgeting?}}


\frame{
    \frametitle{An Example}
    \begin{itemize}\itemsep1em
        \item Telephone survey ($n=1000$)
        \item Simple SRS telephone number sampling frame
        \item Interview takes 15 minutes
        \item Ten respondents paid 200kr (lottery)
    \end{itemize}
}

\frame{
    \frametitle{Fixed costs}
    \begin{itemize}\itemsep1em
        \item<2-> Buying sampling frame: 0kr
        \item<2-> Sampling plan: 0kr
        \item<2-> Staff salaries (200kr/hour): 200 * 150 hours = 30.000kr
        \item<2-> Interviewer training:
            \begin{itemize}
            \item<2-> Trainer (8 hours): 1.400kr
            \item<2-> Interviewers (5 IWRs x 2 hours): 2.000kr
            \end{itemize}
        \item Lottery (10 x 200kr): 2.000kr
    \end{itemize}
}

\frame{
    \frametitle{Variable (per-attempt) costs}
    \begin{itemize}\itemsep1em
        \item<2-> Interviewer salary (6min * 200kr/hour): 20kr/attempt
        \item<2-> No other costs
    \end{itemize}
}

\frame{
    \frametitle{Variable (per-interview) costs}
    \begin{itemize}\itemsep1em
        \item<2-> Interviewer salary (15min * 200kr/hour): 50kr/interview
        \item<2-> Questionnaire materials: 25kr/interview
    \end{itemize}
}

\frame{
    \frametitle{Incidental costs}
    \begin{itemize}\itemsep1em
        \item<2-> Contingency for computer failure: 10.000kr
    \end{itemize}
}

\frame{
    \frametitle{Add it all up}
    \begin{itemize}\itemsep1em
        \item Fixed costs: 33.400 kr
        \item Per-attempt costs: 20kr/attempt
        \item Per-interview costs: 75kr/complete IW
        \item Incidental costs: 10.000kr
    \end{itemize}
}
