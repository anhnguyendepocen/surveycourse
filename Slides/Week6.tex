\frame{
	\frametitle{Survey Mode}
	\begin{itemize}\itemsep2em
		\item Definition?
		\item<2-> Manner in which survey responses are collected
		\item<3-> Not necessarily related to sampling frame
		\item<4-> A survey can use multiple modes
	\end{itemize}
}

\frame{
	\frametitle{Modes}
	\begin{itemize}\itemsep1em
		\item Personal interviewing
		\begin{itemize}
            \item Traditional personal interviewing
    		\item Computer-assisted personal interviewing (CAPI)
		\end{itemize}
		\item Self-administered
    	\begin{itemize}
            \item Paper-and-pencil
    		\item Computer-assisted self-interview (CASI)
    		\item Interactive Voice Response (IVR)
    		\item Telephone-Audio Computer Assisted Self-Interviewing (T-ACASI)
    		\item Web-based
        \end{itemize}
        \item New modes
        \begin{itemize}
            \item Web-based IVR or avatars
            \item Mobile
            \item SMS
        \end{itemize}
	\end{itemize}
}

\section{Costs and Budgeting}
\frame{\tableofcontents[currentsection]}

\frame{
	\frametitle{Three Types of Costs}
	\begin{itemize}\itemsep2em
    	\item Fixed costs
    	\item Variable costs
    	\item Incidental costs (contingencies)
	\end{itemize}
}

\frame{
	\frametitle{Fixed Costs}
	\begin{itemize}\itemsep2em
    	\item Regardless of sample size
    	\item Examples
        	\begin{itemize}
            	\item Cost of creating/finding/building frame
            	\item Sampling plan development
            	\item Staff salaries
            	\item Infrastructure
            	\item Questionnaire development
            	\item Interviewer training
            	\item Data cleaning and analysis
        	\end{itemize}
	\end{itemize}
}

\frame{
	\frametitle{Variable Costs}
	\begin{itemize}\itemsep2em
    	\item Depends on sample size and response rate
        	\begin{enumerate}
            	\item Per-attempt
            	\item Per-interview
        	\end{enumerate}
    	\item Examples
        	\begin{itemize}
            	\item Interviewer wage
            	\item Postage
            	\item Materials (e.g., questionnaires)
            	\item Travel time
            	\item Respondent incentives
        	\end{itemize}
	\end{itemize}
}

\frame{
	\frametitle{Incidental Costs}
	\begin{itemize}\itemsep2em
    	\item Contingency planning (may not spend)
    	\item Examples
        	\begin{itemize}
            	\item Weather
            	\item Lost or stolen data or materials
            	\item Technology failures
        	\end{itemize}
	\end{itemize}
}
