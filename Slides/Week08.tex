\section{Survey Evaluation}
\frame{\tableofcontents[currentsection]}

\frame{
	\frametitle{What is survey evaluation?}
	\begin{itemize}\itemsep1em
		\item Systematic testing of a survey instrument prior to data collection
	\end{itemize}
}

\frame{
	\frametitle{Why pilot test?}
	\begin{itemize}\itemsep2em
		\item<2-> Want to collect accurate, high-quality data
		\item<2-> Survey design is challenging to get right
		\item<2-> Surveys are expensive
		\item<2-> We may not be knowledgeable about the survey topic or population
	\end{itemize}
}

\frame{
	\frametitle{What can we learn from evaluation?}
	\begin{itemize}\itemsep1em
		\item<2-> Determine how to ask questions
		\item<2-> Determine how \textit{not} to ask questions
		\item<2-> Identify confusing questions or answer options
		\item<2-> Identify patterns of bias
		\item<2-> Identify sensitive items 
		\item<2-> Identify context effects
		\item<2-> Identify mode effects
		\item<2-> Estimate survey duration
		\item<2-> Estimate and improve nonresponse \& refusal rates
	\end{itemize}
}

\frame{
	\frametitle{Methods of evaluation}
	\begin{itemize}\itemsep1em
		\item Informal discussion
		\item Expert and/or peer review
		\item Small-\textit{n} pilot tests
		\begin{itemize}
			\item Record checking
			\item Questionnaire experiments
		\end{itemize}
		\item Focus groups
		\item Cognitive interviewing
		\item Interviewer training
	\end{itemize}
}

\frame{
	\frametitle{Informal Discussion}
	\begin{itemize}\itemsep2em
		\item This is what we've been doing
		\item Good starting point
		\item Not systematic
	\end{itemize}
}

\frame{
	\frametitle{Expert/peer review}
	\begin{itemize}\itemsep2em
		\item Identify subject area experts
		\item Ask them to review the design and instrument
		\item Revise based on expert feedback
	\end{itemize}
}
\frame{
	\frametitle{Pilot tests}
	\begin{itemize}\itemsep2em
		\item Administer questionnaire to small sample
		\item Not necessarily random sample
		\item Might use sample for other types of evaluation
			\begin{itemize}
				\item Focus group
				\item Cognitive interviewing
			\end{itemize}
	\end{itemize}
}
\frame{
	\frametitle{Record checking}
	\begin{itemize}\itemsep2em
		\item Check responses from a pilot survey against administrative records
		\item Useful for recall-type questions
	\end{itemize}
}
\frame{
	\frametitle{Questionnaire experiments}
	\begin{itemize}\itemsep2em
		\item Randomly assign respondents to different:
			\begin{itemize}
				\item Individual items
				\item Batteries of items
				\item Entire questionnaires
				\item Modes
			\end{itemize}
		\item Compare results and use ``best'' methods
	\end{itemize}
}
\frame{
	\frametitle{Focus groups}
	\begin{itemize}\itemsep2em
		\item Use before \textit{or} after a questionnaire is drafted
		\item Multiple possible uses
			\begin{itemize}
				\item Develop knowledge about subject area or population
				\item Identify topics to include in survey
				\item Draft question wordings
				\item Clarify question and response meaning
				\item Etc.
			\end{itemize}
	\end{itemize}
}
\frame{
	\frametitle{Cognitive interviewing}
	\begin{itemize}\itemsep2em
		\item Conduct mock interviews
		\item Instead of recording answers, record respondents':
			\begin{itemize}
				\item Interpretations of questions and answers
				\item Difficulties understanding items or topics
				\item Concerns with items or topics
				\item Sensitivity
				\item Etc.
			\end{itemize}
		\item Use quantitative and qualitative analyses
	\end{itemize}
}
\frame{
	\frametitle{Interviewer Training}
	\begin{itemize}\itemsep2em
		\item Train interviewers how to conduct the interview
		\item Use their prior knowledge and experience from the field to update design
		\item We'll talk more about this next week
	\end{itemize}
}
